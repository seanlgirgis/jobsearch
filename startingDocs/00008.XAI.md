# src/ai_tailor.py
import os
from dotenv import load_dotenv
from xai import GrokClient  # adjust to actual import name in 2026

load_dotenv()

class GrokTailor:
    def __init__(self):
        api_key = os.getenv("XAI_API_KEY")
        if not api_key:
            raise ValueError("XAI_API_KEY not found in environment")
        self.client = GrokClient(api_key=api_key)  # or whatever the client class is

    def generate_tailored_section(
        self,
        section_type: str,          # e.g. "summary", "bullets", "cover_letter"
        job_description: str,
        master_data: dict,          # from MasterProfile.career or .skills
        max_tokens: int = 400,
        temperature: float = 0.5
    ) -> str:
        # Build prompt dynamically based on section_type
        prompt = self._build_prompt(section_type, job_description, master_data)
        
        response = self.client.chat.completions.create(
            model="grok-4",  # or latest best model
            messages=[{"role": "user", "content": prompt}],
            temperature=temperature,
            max_tokens=max_tokens,
        )
        return response.choices[0].message.content.strip()

    def _build_prompt(self, section_type, job_desc, master_data):
        # Example for summary â€” we'll expand for other types
        if section_type == "summary":
            return f"""
You are an expert ATS-optimized resume writer.
Using this master career data:
{self._summarize_master(master_data)}

And this job description:
{job_desc}

Write a 4-6 line professional summary that:
- Leads with strongest matching title/expertise
- Incorporates exact keywords from the job
- Quantifies achievements where possible
- Sounds natural and confident
"""
        # Add cases for "experience_bullets", "cover_letter", etc.
        raise NotImplementedError(f"Section {section_type} not implemented yet")

    def _summarize_master(self, data):
        # Simple string summary of relevant parts â€” expand as needed
        return f"Name: {data.get('personal', {}).get('name')}\nExperience: {len(data.get('experience', []))} roles\nSkills: top ones include Python 15y, SQL/Oracle 18y, AWS, PySpark, GenAI..."
