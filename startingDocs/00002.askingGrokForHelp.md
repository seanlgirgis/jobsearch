### Recommended Approach: Simple Python Web App with Streamlit
- **Why Streamlit?** 
  - It's the fastest way to build a clean, interactive web app with almost no frontend code.
  - Perfect for forms, text input, buttons, and displaying generated resumes/cover letters.
  - Runs locally on your computer first, then you can deploy it for free (e.g., on Streamlit Community Cloud).
  - Easy integration with the xAI/Grok API (it's OpenAI-compatible).

Tech stack:
- Python (you probably already have it or can install easily)
- Streamlit (one-line install)
- xAI/Grok API
- Simple local storage first (JSON file or SQLite), later optional cloud database

### Let's Build It — Starter Code
Install these first (in your terminal):
```bash
pip install streamlit openai python-dotenv
```

Create a file `resume_app.py`:

```python
import streamlit as st
import json
import os
from openai import OpenAI
from dotenv import load_dotenv

load_dotenv()

# xAI API setup — uses OpenAI library but points to xAI
client = OpenAI(
    api_key=os.getenv("XAI_API_KEY"),  # put your key in .env file
    base_url="https://api.x.ai/v1"
)

DATA_FILE = "my_career_data.json"

# Load or initialize data
if os.path.exists(DATA_FILE):
    with open(DATA_FILE, "r") as f:
        data = json.load(f)
else:
    data = {
        "personal": {"name": "", "email": "", "phone": "", "summary": ""},
        "experience": [],
        "education": [],
        "skills": {"technical": [], "soft": [], "tools": []},
        "projects": []
    }

def save_data():
    with open(DATA_FILE, "w") as f:
        json.dump(data, f, indent=4)

st.title("My AI Resume & Cover Letter Builder")

page = st.sidebar.selectbox("Section", ["My Profile", "Generate", "History"])

if page == "My Profile":
    st.header("Edit Your Career Data")
    
    # Personal
    data["personal"]["name"] = st.text_input("Name", data["personal"]["name"])
    data["personal"]["email"] = st.text_input("Email", data["personal"]["email"])
    data["personal"]["phone"] = st.text_input("Phone", data["personal"]["phone"])
    data["personal"]["summary"] = st.text_area("Professional Summary", data["personal"]["summary"], height=150)
    
    # Experience (add multiple)
    st.subheader("Work Experience")
    for i, exp in enumerate(data["experience"]):
        with st.expander(f"Job {i+1}: {exp.get('role', '')} at {exp.get('company', '')}"):
            exp["company"] = st.text_input("Company", exp.get("company", ""), key=f"comp{i}")
            exp["role"] = st.text_input("Role", exp.get("role", ""), key=f"role{i}")
            exp["dates"] = st.text_input("Dates", exp.get("dates", ""), key=f"dates{i}")
            exp["bullets"] = st.text_area("Bullet points (one per line)", "\n".join(exp.get("bullets", [])), key=f"bullets{i}")
            data["experience"][i] = exp
    if st.button("Add New Job"):
        data["experience"].append({"company": "", "role": "", "dates": "", "bullets": []})
    
    # Similar sections for Education, Skills, Projects (you can copy-paste pattern)
    
    if st.button("Save All Data"):
        save_data()
        st.success("Data saved!")

elif page == "Generate":
    st.header("Generate Resume / Cover Letter")
    job_desc = st.text_area("Paste the Job Description here", height=300)
    
    if st.button("Generate Resume") and job_desc:
        with st.spinner("Thinking..."):
            prompt = f"""
            Using this career data: {json.dumps(data, indent=2)}
            Create a professional one-page resume tailored to this job: {job_desc}
            Format in clean Markdown with sections: Contact, Summary, Experience, Education, Skills.
            """
            response = client.chat.completions.create(
                model="grok-4",  # or "grok-3" if that's what you have
                messages=[{"role": "user", "content": prompt}],
                temperature=0.7
            )
            resume = response.choices[0].message.content
            st.markdown(resume)
            st.download_button("Download as Markdown", resume, file_name="resume.md")
    
    if st.button("Generate Cover Letter") and job_desc:
        with st.spinner("Writing..."):
            prompt = f"""
            Using this career data: {json.dumps(data, indent=2)}
            Write a compelling 3-4 paragraph cover letter for this job: {job_desc}
            Make it specific, enthusiastic, and highlight relevant achievements.
            """
            response = client.chat.completions.create(
                model="grok-4",
                messages=[{"role": "user", "content": prompt}],
                temperature=0.7
            )
            letter = response.choices[0].message.content
            st.markdown(letter)
            st.download_button("Download as Markdown", letter, file_name="cover_letter.md")

# Add History page later to list past generations
```
